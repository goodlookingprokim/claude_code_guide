1. 환경변수 설정

먼저 필요한 환경변수들을 설정합니다:

.env 파일 생성 
# GitHub 설정 (필수)
export GITHUB_TOKEN=ghp_xxxxxxxxxxxxxxxxxxxx

# Supabase 설정 (필수)
export SUPABASE_URL=https://your-project.supabase.co
export SUPABASE_ACCESS_TOKEN=sbp_xxxxxxxxxxxxxxxxxxxx

# DigitalOcean 설정 (필수)
export DIGITALOCEAN_API_TOKEN=dop_v1_xxxxxxxxxxxxxxxxxxxx

# Notion 설정 (선택사항)
export NOTION_API_KEY=secret_xxxxxxxxxxxxxxxxxxxx

# Slack 설정 (선택사항)
export SLACK_BOT_TOKEN=xoxb-xxxxxxxxxxxxxxxxxxxx

# 데이터베이스 설정 (선택사항)
export DATABASE_URL=postgresql://user:password@localhost:5432/dbname

# OpenAI 설정 (선택사항)
export OPENAI_API_KEY=sk-xxxxxxxxxxxxxxxxxxxx

# 기타 설정
export ALLOWED_DIRS="$HOME/Projects,$HOME/Documents"

2. 원클릭 설치 스크립트

모든 MCP 서버를 한번에 설치하는 스크립트입니다:

setup_mcp_stack.sh 
#!/bin/bash

echo "🚀 Claude Code MCP Stack 설정을 시작합니다..."

# 환경변수 확인 함수
check_env_var() {
    if [ -z "${!1}" ]; then
        echo "❌ $1 환경변수가 설정되지 않았습니다."
        echo "💡 export $1=your_token_here 로 설정해주세요."
        return 1
    fi
    return 0
}

# 필수 환경변수 확인
echo "🔍 환경변수 확인 중..."
ENV_CHECK_FAILED=false

if ! check_env_var "GITHUB_TOKEN"; then ENV_CHECK_FAILED=true; fi
if ! check_env_var "SUPABASE_ACCESS_TOKEN"; then ENV_CHECK_FAILED=true; fi
if ! check_env_var "DIGITALOCEAN_API_TOKEN"; then ENV_CHECK_FAILED=true; fi

if [ "$ENV_CHECK_FAILED" = true ]; then
    echo ""
    echo "📝 환경변수 설정 예시:"
    echo "export GITHUB_TOKEN=ghp_xxxxxxxxxxxxxxxxxxxx"
    echo "export SUPABASE_ACCESS_TOKEN=sbp_xxxxxxxxxxxxxxxxxxxx"
    echo "export DIGITALOCEAN_API_TOKEN=dop_v1_xxxxxxxxxxxxxxxxxxxx"
    echo ""
    echo "🔗 토큰 발급 방법은 아래 가이드를 참고하세요."
    exit 1
fi

echo "✅ 환경변수 확인 완료!"

# 1. 필수 개발 도구
echo ""
echo "📚 필수 개발 도구 MCP 서버 설치 중..."

claude mcp add github \
  -- npx -y @modelcontextprotocol/server-github \
  --github-token="$GITHUB_TOKEN"

claude mcp add filesystem \
  -- npx -y @modelcontextprotocol/server-filesystem \
  --allowed-dirs="$HOME/Projects,$HOME/Documents"

claude mcp add sqlite \
  -- npx -y @modelcontextprotocol/server-sqlite \
  --db-path="./development.db"

# 2. 클라우드 & 인프라
echo ""
echo "☁️ 클라우드 & 인프라 MCP 서버 설치 중..."

claude mcp add supabase \
  -- npx -y @supabase/mcp-server-supabase@latest \
  --access-token="$SUPABASE_ACCESS_TOKEN"

claude mcp add digitalocean \
  -- env DIGITALOCEAN_API_TOKEN="$DIGITALOCEAN_API_TOKEN" \
  npx -y @digitalocean/mcp-server

# 3. 웹 자동화 & 스크래핑
echo ""
echo "🕷️ 웹 자동화 MCP 서버 설치 중..."

claude mcp add puppeteer \
  -- npx -y @modelcontextprotocol/server-puppeteer

claude mcp add playwright \
  -- npx -y @playwright/mcp-server

# 4. 생산성 도구 (선택사항)
if [ ! -z "$NOTION_API_KEY" ]; then
    echo ""
    echo "📊 생산성 도구 MCP 서버 설치 중..."
    
    claude mcp add notion \
      -- env NOTION_API_KEY="$NOTION_API_KEY" \
      npx -y @notion/mcp-server
fi

if [ ! -z "$SLACK_BOT_TOKEN" ]; then
    claude mcp add slack \
      -- env SLACK_BOT_TOKEN="$SLACK_BOT_TOKEN" \
      npx -y @slack/mcp-server
fi

# 5. 데이터 분석
echo ""
echo "📈 데이터 분석 MCP 서버 설치 중..."

if [ ! -z "$DATABASE_URL" ]; then
    claude mcp add postgres \
      -- env DATABASE_URL="$DATABASE_URL" \
      npx -y @modelcontextprotocol/server-postgres
fi

claude mcp add memory \
  -- npx -y @modelcontextprotocol/server-memory

echo ""
echo "🎉 MCP Stack 설정이 완료되었습니다!"
echo "📋 설치된 서버 목록을 확인하려면: claude mcp list"

source .env

2. TypeScript MCP 서버

src/index.ts 
#!/usr/bin/env node

import { Server } from '@modelcontextprotocol/sdk/server/index.js';
import { StdioServerTransport } from '@modelcontextprotocol/sdk/server/stdio.js';
import {
  CallToolRequestSchema,
  ListToolsRequestSchema,
} from '@modelcontextprotocol/sdk/types.js';

// 커스텀 도구 정의
const server = new Server(
  {
    name: 'my-custom-mcp-server',
    version: '1.0.0',
  },
  {
    capabilities: {
      tools: {},
    },
  }
);

// 도구 목록
server.setRequestHandler(ListToolsRequestSchema, async () => {
  return {
    tools: [
      {
        name: 'custom_tool',
        description: '커스텀 도구 설명',
        inputSchema: {
          type: 'object',
          properties: {
            input: {
              type: 'string',
              description: '입력 파라미터',
            },
          },
          required: ['input'],
        },
      },
    ],
  };
});

// 도구 실행
server.setRequestHandler(CallToolRequestSchema, async (request) => {
  const { name, arguments: args } = request.params;

  if (name === 'custom_tool') {
    const input = args?.input as string;
    
    // 커스텀 로직 구현
    const result = `처리 결과: ${input}`;
    
    return {
      content: [
        {
          type: 'text',
          text: result,
        },
      ],
    };
  }

  throw new Error(`Unknown tool: ${name}`);
});

// 서버 시작
async function main() {
  const transport = new StdioServerTransport();
  await server.connect(transport);
  console.error('커스텀 MCP 서버가 시작되었습니다.');
}

main().catch((error) => {
  console.error('서버 시작 실패:', error);
  process.exit(1);
});

📊 관리 및 모니터링

MCP 서버 관리 스크립트

mcp_manager.sh 
#!/bin/bash

# MCP 서버 목록 확인
show_servers() {
    echo "📋 설치된 MCP 서버 목록:"
    claude mcp list
}

# 서버 상태 확인
check_health() {
    echo "🔍 MCP 서버 상태 확인 중..."
    
    servers=("github" "filesystem" "sqlite" "supabase" "digitalocean" "puppeteer")
    
    for server in "${servers[@]}"; do
        if claude mcp status "$server" &>/dev/null; then
            echo "✅ $server: 정상"
        else
            echo "❌ $server: 오류"
        fi
    done
}

# 서버 재시작
restart_server() {
    if [ -z "$1" ]; then
        echo "❌ 서버 이름을 지정해주세요."
        echo "사용법: restart_server "
        return 1
    fi
    
    echo "🔄 $1 서버 재시작 중..."
    claude mcp remove "$1"
    # 여기에 해당 서버의 추가 명령어 실행
    echo "✅ $1 서버가 재시작되었습니다."
}

# 로그 확인
show_logs() {
    if [ -z "$1" ]; then
        echo "❌ 서버 이름을 지정해주세요."
        return 1
    fi
    
    echo "📋 $1 서버 로그:"
    claude mcp logs "$1"
}

# 서버 설정 백업
backup_config() {
    echo "💾 MCP 설정 백업 중..."
    claude mcp export > mcp_backup_$(date +%Y%m%d_%H%M%S).json
    echo "✅ 백업이 완료되었습니다."
}

# 메뉴
case "$1" in
    "list")
        show_servers
        ;;
    "health")
        check_health
        ;;
    "restart")
        restart_server "$2"
        ;;
    "logs")
        show_logs "$2"
        ;;
    "backup")
        backup_config
        ;;
    *)
        echo "🚀 MCP 서버 관리 도구"
        echo ""
        echo "사용법:"
        echo "  $0 list     - 서버 목록 확인"
        echo "  $0 health   - 서버 상태 확인"
        echo "  $0 restart  - 서버 재시작"
        echo "  $0 logs     - 서버 로그 확인"
        echo "  $0 backup   - 설정 백업"
        ;;
esac
